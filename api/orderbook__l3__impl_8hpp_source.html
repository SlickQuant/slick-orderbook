14SLICK_OB_INLINE OrderBookL3::OrderBookL3(SymbolId symbol,
15 std::size_t initial_order_capacity,
16 std::size_t initial_level_capacity)
17 : symbol_(symbol),
18 order_map_(initial_order_capacity),
19 order_pool_(initial_order_capacity),
20 last_seq_num_(0) {
21 // Initialize level maps with comparators
22 // Note: Clang-17 with GCC-14 libstdc++ has issues with aggregate initialization
23 // Use constructor syntax instead of brace initialization to avoid C++23 tuple conversion issues
24 levels_[Side::Buy] = PriceLevelMap(PriceComparator(Side::Buy));
25 levels_[Side::Sell] = PriceLevelMap(PriceComparator(Side::Sell));
26
27 // Reserve capacity for price levels (flat_map doesn't have reserve in C++23)
28 // This will be used during insertions to pre-allocate space
29 (void)initial_level_capacity; // Mark as used
30
31 // Initialize cached top-of-book
32 cached_tob_.symbol = symbol_;
33}

40SLICK_OB_INLINE bool OrderBookL3::addOrModifyOrder(OrderId order_id, Side side, Price price, Quantity quantity,
41 Timestamp timestamp, uint64_t priority, uint64_t seq_num, bool is_last_in_batch) {
42 SLICK_ASSERT(side < SideCount);
43
44 // Validate sequence number - reject out-of-order (seq_num < last_seq_num)
45 if (seq_num > 0) {
46 if (seq_num < last_seq_num_) {
47 // Out of order - reject
48 return false;
49 }
50 last_seq_num_ = seq_num;
51 }
52
53 // Check if order already exists
54 detail::Order* order = order_map_.find(order_id);
55 if (order) [[unlikely]] {
56 if (order->side != side) {
57 // Side mismatch implies OrderId reuse - reject.
58 SLICK_ASSERT(false);
59 return false;
60 }
61
62 if (SLICK_UNLIKELY(quantity <= 0)) {
63 return (quantity == 0) ? deleteOrder(order_id, seq_num, is_last_in_batch) : false;
64 }
65
66 // Idempotent update - nothing to do.
67 if (order->price == price && order->quantity == quantity) {
68 return true;
69 }
70
71 // Use the provided timestamp for update notifications.
72 order->timestamp = timestamp;
73 return modifyOrder(order_id, price, quantity, seq_num, is_last_in_batch);
74 }
75
76 if (SLICK_UNLIKELY(quantity <= 0)) {
77 return false;
78 }
79
80 // Allocate order from pool
81 order = order_pool_.construct(order_id, price, quantity, side, timestamp, priority);
82 if (SLICK_UNLIKELY(order == nullptr)) {
83 return false;
84 }
85
86 // Get or create price level
87 auto [level, level_idx, is_new] = getOrCreateLevel(side, price);
88
89 // track starting index
90 change_starting_index_ = std::min<uint16_t>(change_starting_index_, level_idx);
91
92 // Insert order into level (maintains priority order)
93 level->insertOrder(order);
94
95 // Add to order map
96 order_map_.insert(order);
97
98 // New order: both price and quantity changed
99 uint8_t order_flags = PriceChanged | QuantityChanged;
100 uint8_t level_change_flag = QuantityChanged;
101 if (is_new) {
102 level_change_flag |= PriceChanged;
103 }
104 // Add LastInBatch flag if this is the last update
105 if (is_last_in_batch) {
106 order_flags |= LastInBatch;
107 level_change_flag |= LastInBatch;
108 }
109
110 // Notify observers
111 notifyOrderUpdate(order, 0, 0, timestamp, level_idx, order_flags, seq_num);
112 notifyPriceLevelUpdate(side, price, level->getTotalQuantity(), timestamp, level_idx, level_change_flag, seq_num);
113 if (change_starting_index_ == 0 && is_last_in_batch) {
114 notifyTopOfBookIfChanged(timestamp);
115 change_starting_index_ = INVALID_INDEX;
116 }
117
118 return true;
119}

121SLICK_OB_INLINE bool OrderBookL3::addOrder(OrderId order_id, Side side, Price price, Quantity quantity,
122 Timestamp timestamp, uint64_t priority, uint64_t seq_num, bool is_last_in_batch) {
123 // Validate sequence number - reject out-of-order (seq_num < last_seq_num)
124 if (seq_num > 0) {
125 if (seq_num < last_seq_num_) {
126 // Out of order - reject
127 return false;
128 }
129 last_seq_num_ = seq_num;
130 }
131
132 // Strict add - fail if order already exists
133 if (order_map_.contains(order_id)) {
134 return false;
135 }
136 // Use timestamp as priority if not specified
137 uint64_t actual_priority = (priority == 0) ? timestamp : priority;
138 return addOrModifyOrder(order_id, side, price, quantity, timestamp, actual_priority, seq_num, is_last_in_batch);
139}

141SLICK_OB_INLINE bool OrderBookL3::modifyOrder(OrderId order_id, Price new_price, Quantity new_quantity,
142 uint64_t seq_num, bool is_last_in_batch) {
143 // Validate sequence number - reject out-of-order (seq_num < last_seq_num)
144 if (seq_num > 0) {
145 if (seq_num < last_seq_num_) {
146 // Out of order - reject
147 return false;
148 }
149 last_seq_num_ = seq_num;
150 }
151
152 // Find order
153 detail::Order* order = order_map_.find(order_id);
154 if (!order) {
155 return false;
156 }
157
158 if (SLICK_UNLIKELY(new_quantity < 0)) {
159 return false;
160 }
161
162 // Check if this is a delete (quantity = 0)
163 if (new_quantity == 0) {
164 return deleteOrder(order_id, seq_num, is_last_in_batch);
165 }
166
167 const Timestamp timestamp = order->timestamp; // Use original timestamp for modifications
168 const Price old_price = order->price;
169 const Quantity old_quantity = order->quantity;
170 const Side side = order->side;
171
172 // Check what changed
173 const bool price_changed = (new_price != old_price);
174 const bool quantity_changed = (new_quantity != old_quantity);
175
176 if (!price_changed && !quantity_changed) {
177 // No change
178 return true;
179 }
180
181 if (price_changed) {
182 // Price changed - remove from old level (if exists) and add to new level
183 auto [old_level, old_level_idx] = getLevel(side, old_price);
184 Quantity old_level_total = 0;
185
186 if (old_level) {
187 // track starting index
188 change_starting_index_ = std::min<uint16_t>(change_starting_index_, old_level_idx);
189
190 // Remove from old level
191 old_level->removeOrder(order);
192 old_level_total = old_level->getTotalQuantity();
193
194 uint8_t old_level_change_flags = QuantityChanged;
195 if (removeLevelIfEmpty(side, old_price)) {
196 old_level_change_flags |= PriceChanged;
197 }
198 // Don't add LastInBatch to intermediate old level update
199 notifyPriceLevelUpdate(side, old_price, old_level_total, timestamp, old_level_idx, old_level_change_flags, seq_num);
200 }
201
202 // Update order fields
203 order->price = new_price;
204 order->quantity = new_quantity;
205
206 // Get or create new level
207 auto [new_level, new_level_idx, is_new] = getOrCreateLevel(side, new_price);
208
209 change_starting_index_ = std::min<uint16_t>(change_starting_index_, new_level_idx);
210
211 // Insert into new level
212 new_level->insertOrder(order);
213
214 // Notify observers (price changed: both price and quantity flags)
215 uint8_t order_flags = PriceChanged;
216 if (quantity_changed) {
217 order_flags |= QuantityChanged;
218 }
219 uint8_t new_level_change_flags = QuantityChanged;
220 if (is_new) {
221 new_level_change_flags |= PriceChanged;
222 }
223 // Add LastInBatch flag if this is the last update
224 if (is_last_in_batch) {
225 order_flags |= LastInBatch;
226 new_level_change_flags |= LastInBatch;
227 }
228
229 notifyOrderUpdate(order, old_quantity, old_price, timestamp, new_level_idx, order_flags, seq_num);
230 notifyPriceLevelUpdate(side, new_price, new_level->getTotalQuantity(), timestamp, new_level_idx, new_level_change_flags, seq_num);
231 if (change_starting_index_ == 0 && is_last_in_batch) {
232 notifyTopOfBookIfChanged(timestamp);
233 change_starting_index_ = INVALID_INDEX;
234 }
235
236 } else {
237 // Only quantity changed
238 auto [level, level_index, is_new] = getOrCreateLevel(side, old_price);
239
240 change_starting_index_ = std::min<uint16_t>(change_starting_index_, level_index);
241
242 // Update quantity
243 level->updateOrderQuantity(old_quantity, new_quantity);
244 order->quantity = new_quantity;
245
246 // Notify observers (only quantity changed)
247 uint8_t order_flags = QuantityChanged;
248 uint8_t level_change_flags = QuantityChanged;
249 if (is_new) [[unlikely]] {
250 level_change_flags |= PriceChanged;
251 }
252 // Add LastInBatch flag if this is the last update
253 if (is_last_in_batch) {
254 order_flags |= LastInBatch;
255 level_change_flags |= LastInBatch;
256 }
257
258 notifyOrderUpdate(order, old_quantity, old_price, timestamp, level_index, order_flags, seq_num);
259 notifyPriceLevelUpdate(side, old_price, level->getTotalQuantity(), timestamp, level_index, level_change_flags, seq_num);
260 if (change_starting_index_ == 0 && is_last_in_batch) {
261 notifyTopOfBookIfChanged(timestamp);
262 change_starting_index_ = INVALID_INDEX;
263 }
264 }
265
266 return true;
267}

269SLICK_OB_INLINE bool OrderBookL3::deleteOrder(OrderId order_id, uint64_t seq_num, bool is_last_in_batch) noexcept {
270 // Validate sequence number - reject out-of-order (seq_num < last_seq_num)
271 if (seq_num > 0) {
272 if (seq_num < last_seq_num_) {
273 // Out of order - reject
274 return false;
275 }
276 last_seq_num_ = seq_num;
277 }
278
279 // Find order
280 detail::Order* order = order_map_.find(order_id);
281 if (!order) {
282 return false;
283 }
284
285 const Timestamp timestamp = order->timestamp;
286 const Price price = order->price;
287 const Side side = order->side;
288
289 // Get level
290 auto [level, level_idx] = getLevel(side, price);
291 if (!level) {
292 // Level doesn't exist - data structure inconsistency
293 // Deletion: both price and quantity changed
294 uint8_t order_flags = PriceChanged | QuantityChanged;
295 if (is_last_in_batch) {
296 order_flags |= LastInBatch;
297 }
298 // Notify deletion, then clean up (use max index since level not found)
299 notifyOrderDelete(order, timestamp, std::numeric_limits<uint16_t>::max(), order_flags, seq_num);
300 order_map_.erase(order_id);
301 order_pool_.destroy(order);
302 return false;
303 }
304
305 change_starting_index_ = std::min<uint16_t>(change_starting_index_, level_idx);
306
307 // Remove from level
308 level->removeOrder(order);
309 const Quantity level_total = level->getTotalQuantity();
310
311 // Remove from order map
312 order_map_.erase(order_id);
313
314 // Deletion: both price and quantity changed
315 uint8_t order_flags = PriceChanged | QuantityChanged;
316 uint8_t level_change_flags = QuantityChanged;
317 if (removeLevelIfEmpty(side, price)) {
318 level_change_flags |= PriceChanged;
319 }
320 // Add LastInBatch flag if this is the last update
321 if (is_last_in_batch) {
322 order_flags |= LastInBatch;
323 level_change_flags |= LastInBatch;
324 }
325
326 // Notify observers (before destroying order)
327 notifyOrderDelete(order, timestamp, level_idx, order_flags, seq_num);
328 notifyPriceLevelUpdate(side, price, level_total, timestamp, level_idx, level_change_flags, seq_num);
329
330 // Destroy order
331 order_pool_.destroy(order);
332
333 // Notify ToB if changed
334 if (change_starting_index_ == 0 && is_last_in_batch) {
335 notifyTopOfBookIfChanged(timestamp);
336 change_starting_index_ = INVALID_INDEX;
337 }
338
339 return true;
340}

342SLICK_OB_INLINE bool OrderBookL3::executeOrder(OrderId order_id, Quantity executed_quantity, uint64_t seq_num, bool is_last_in_batch) {
343 // Validate sequence number - reject out-of-order (seq_num < last_seq_num)
344 if (seq_num > 0) {
345 if (seq_num < last_seq_num_) {
346 // Out of order - reject
347 return false;
348 }
349 last_seq_num_ = seq_num;
350 }
351
352 detail::Order* order = order_map_.find(order_id);
353 if (!order) {
354 return false;
355 }
356
357 if (SLICK_UNLIKELY(executed_quantity <= 0 || executed_quantity > order->quantity)) {
358 SLICK_ASSERT(executed_quantity > 0 && executed_quantity <= order->quantity);
359 return false;
360 }
361
362 const Quantity remaining = order->quantity - executed_quantity;
363
364 if (remaining == 0) {
365 // Fully executed - delete order (pass through seq_num and is_last_in_batch)
366 return deleteOrder(order_id, seq_num, is_last_in_batch);
367 } else {
368 // Partial execution - reduce quantity (pass through seq_num and is_last_in_batch)
369 return modifyOrder(order_id, order->price, remaining, seq_num, is_last_in_batch);
370 }
371}

377SLICK_OB_INLINE const detail::PriceLevelL3* OrderBookL3::getBestBid() const noexcept {
378 const auto& bids = levels_[Side::Buy];
379 if (bids.empty()) {
380 return nullptr;
381 }
382 // FlatMap is sorted by key (Price) in ascending order by default
383 // For Buy side, we want highest price first, so use reverse iterator
384 return &bids.begin()->second;

393 return &asks.begin()->second;

396SLICK_OB_INLINE TopOfBook OrderBookL3::getTopOfBook() const noexcept {
397 const auto* bid = getBestBid();
398 const auto* ask = getBestAsk();
399
400 TopOfBook tob;
401 tob.symbol = symbol_;
402
403 if (bid) {
404 tob.best_bid = bid->price;
405 tob.bid_quantity = bid->getTotalQuantity();
406 if (!bid->orders.empty()) {
407 tob.timestamp = bid->orders.front()->timestamp;
408 }
409 }
410
411 if (ask) {
412 tob.best_ask = ask->price;
413 tob.ask_quantity = ask->getTotalQuantity();
414 if (!ask->orders.empty()) {
415 const Timestamp ask_time = ask->orders.front()->timestamp;
416 if (ask_time > tob.timestamp) {
417 tob.timestamp = ask_time;
418 }
419 }
420 }
421
422 return tob;
423}

425SLICK_OB_INLINE std::vector<detail::PriceLevelL2> OrderBookL3::getLevelsL2(Side side, std::size_t depth) const {
426 SLICK_ASSERT(side < SideCount);
427
428 const auto& level_map = levels_[side];
429 std::vector<detail::PriceLevelL2> result;
430
431 if (level_map.empty()) {
432 return result;
433 }
434
435 const std::size_t count = (depth == 0) ? level_map.size() : std::min(depth, level_map.size());
436 result.reserve(count);
437
438 if (side == Side::Buy) {
439 // For bids, iterate in reverse (highest price first)
440 auto it = level_map.begin();
441 for (std::size_t i = 0; i < count && it != level_map.end(); ++i, ++it) {
442 const auto& [price, level] = *it;
443 Timestamp latest_timestamp = level.orders.empty() ? 0 : level.orders.front()->timestamp;
444 result.emplace_back(price, level.getTotalQuantity(), latest_timestamp);
445 }
446 } else {
447 // For asks, iterate forward (lowest price first)
448 auto it = level_map.begin();
449 for (std::size_t i = 0; i < count && it != level_map.end(); ++i, ++it) {
450 const auto& [price, level] = *it;
451 Timestamp latest_timestamp = level.orders.empty() ? 0 : level.orders.front()->timestamp;
452 result.emplace_back(price, level.getTotalQuantity(), latest_timestamp);
453 }
454 }
455
456 return result;
457}

515SLICK_OB_INLINE void OrderBookL3::clearSide(Side side) noexcept {
516 SLICK_ASSERT(side < SideCount);
517
518 // Delete all orders on this side
519 auto& level_map = levels_[side];
520 for (auto& [price, level] : level_map) {
521 for (auto it = level.orders.begin(); it != level.orders.end(); ) {
522 auto* order = &(*it);
523 ++it; // Advance before unlinking/destroying.
524 order_map_.erase(order->order_id);
525 level.orders.erase(order);
526 order_pool_.destroy(order);
527 }
528 }
529
530 // Clear level map
531 level_map.clear();
532}

539SLICK_OB_INLINE std::tuple<detail::PriceLevelL3*, uint16_t, bool> OrderBookL3::getOrCreateLevel(Side side, Price price) {
540 SLICK_ASSERT(side < SideCount);
541
542 auto& level_map = levels_[side];
543 auto it = level_map.find(price);
544
545 if (it != level_map.end()) {
546 uint16_t index = static_cast<uint16_t>(std::distance(level_map.begin(), it));
547 return {&it->second, index, false};
548 }
549
550 // Create new level
551 auto [new_it, inserted] = level_map.emplace(price, detail::PriceLevelL3{price});
552 SLICK_ASSERT(inserted);
553 uint16_t index = static_cast<uint16_t>(std::distance(level_map.begin(), new_it));
554 return {&new_it->second, index, true};
555}

557SLICK_OB_INLINE bool OrderBookL3::removeLevelIfEmpty(Side side, Price price) noexcept {
558 SLICK_ASSERT(side < SideCount);
559
560 auto& level_map = levels_[side];
561 auto it = level_map.find(price);
562
563 if (it != level_map.end() && it->second.isEmpty()) {
564 level_map.erase(it);
565 return true;
566 }
567 return false;

570SLICK_OB_INLINE uint16_t OrderBookL3::calculateLevelIndex(Side side, Price price) const noexcept {
571 SLICK_ASSERT(side < SideCount);
572 const auto& level_map = levels_[side];
573
574 // Find the price level using lower_bound
575 auto it = level_map.lower_bound(price);
576
577 // If exact match, calculate its index
578 if (it != level_map.end() && it->first == price) {
579 return static_cast<uint16_t>(std::distance(level_map.begin(), it));
580 }
581
582 // Price not found - return the index where it would be inserted
583 return static_cast<uint16_t>(std::distance(level_map.begin(), it));
584}

651SLICK_OB_INLINE void OrderBookL3::notifyTopOfBookIfChanged(Timestamp timestamp) {
652 // Compute current top-of-book
653 const auto* bid = getBestBid();
654 const auto* ask = getBestAsk();
655
656 Price new_best_bid = bid ? bid->price : 0;
657 Quantity new_bid_qty = bid ? bid->getTotalQuantity() : 0;
658 Price new_best_ask = ask ? ask->price : 0;
659 Quantity new_ask_qty = ask ? ask->getTotalQuantity() : 0;
660
661 // Check if best bid or ask changed
662 const bool bid_changed = (cached_tob_.best_bid != new_best_bid) ||
663 (cached_tob_.bid_quantity != new_bid_qty);
664 const bool ask_changed = (cached_tob_.best_ask != new_best_ask) ||
665 (cached_tob_.ask_quantity != new_ask_qty);
666
667 if (bid_changed || ask_changed) {
668 // Update cached values
669 cached_tob_.best_bid = new_best_bid;
670 cached_tob_.bid_quantity = new_bid_qty;
671 cached_tob_.best_ask = new_best_ask;
672 cached_tob_.ask_quantity = new_ask_qty;
673 cached_tob_.timestamp = timestamp;
674
675 // Notify observers
676 observers_.notifyTopOfBookUpdate(cached_tob_);
677 }
678}

680SLICK_OB_INLINE void OrderBookL3::emitSnapshot(Timestamp timestamp) {
681 observers_.notifySnapshotBegin(symbol_, last_seq_num_, timestamp);
682
683 // Bids: highest price first (best = index 0)
684 uint16_t level_idx = 0;
685 for (auto it = levels_[Side::Buy].begin(); it != levels_[Side::Buy].end(); ++it, ++level_idx) {
686 const auto& level = it->second;
687 for (const auto& order : level.orders) {
688 OrderUpdate update{
689 symbol_,
690 order.order_id,
691 order.side,
692 order.price,
693 order.quantity,
694 timestamp,
695 level_idx,
