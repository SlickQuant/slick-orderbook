42SLICK_OB_INLINE bool OrderBookL3::addOrModifyOrder(OrderId order_id, Side side, Price price, Quantity quantity,
43 Timestamp timestamp, uint64_t priority, uint64_t seq_num, bool is_last_in_batch) {
44 SLICK_ASSERT(side < SideCount);
45
46 // Validate sequence number - reject out-of-order (seq_num < last_seq_num)
47 if (seq_num > 0) {
48 if (seq_num < last_seq_num_) {
49 // Out of order - reject
50 return false;
51 }
52 last_seq_num_ = seq_num;
53 }
54
55 // Check if order already exists
56 detail::Order* order = order_map_.find(order_id);
57 if (order) [[unlikely]] {
58 if (order->side != side) {
59 // Side mismatch implies OrderId reuse - reject.
60 SLICK_ASSERT(false);
61 return false;
62 }
63
64 if (SLICK_UNLIKELY(quantity <= 0)) {
65 return (quantity == 0) ? deleteOrder(order_id, seq_num, is_last_in_batch) : false;
66 }
67
68 // Idempotent update - nothing to do.
69 if (order->price == price && order->quantity == quantity) {
70 return true;
71 }
72
73 // Use the provided timestamp for update notifications.
74 order->timestamp = timestamp;
75 return modifyOrder(order_id, price, quantity, seq_num, is_last_in_batch);
76 }
77
78 if (SLICK_UNLIKELY(quantity <= 0)) {
79 return false;
80 }
81
82 // Allocate order from pool
83 order = order_pool_.construct(order_id, price, quantity, side, timestamp, priority);
84 if (SLICK_UNLIKELY(order == nullptr)) {
85 return false;
86 }
87
88 // Get or create price level
89 auto [level, level_idx, is_new] = getOrCreateLevel(side, price);
90
91 // Insert order into level (maintains priority order)
92 level->insertOrder(order);
93
94 // Add to order map
95 order_map_.insert(order);
96
97 // New order: both price and quantity changed
98 uint8_t order_flags = PriceChanged | QuantityChanged;
99 uint8_t level_change_flag = QuantityChanged;
100 if (is_new) {
101 level_change_flag |= PriceChanged;
102 }
103 // Add LastInBatch flag if this is the last update
104 if (is_last_in_batch) {
105 order_flags |= LastInBatch;
106 level_change_flag |= LastInBatch;
107 }
108
109 // Notify observers
110 notifyOrderUpdate(order, 0, 0, timestamp, level_idx, order_flags, seq_num);
111 notifyPriceLevelUpdate(side, price, level->getTotalQuantity(), timestamp, level_idx, level_change_flag, seq_num);
112 notifyTopOfBookIfChanged(timestamp, order_flags);
113
114 return true;
115}

117SLICK_OB_INLINE bool OrderBookL3::addOrder(OrderId order_id, Side side, Price price, Quantity quantity,
118 Timestamp timestamp, uint64_t priority, uint64_t seq_num, bool is_last_in_batch) {
119 // Validate sequence number - reject out-of-order (seq_num < last_seq_num)
120 if (seq_num > 0) {
121 if (seq_num < last_seq_num_) {
122 // Out of order - reject
123 return false;
124 }
125 last_seq_num_ = seq_num;
126 }
127
128 // Strict add - fail if order already exists
129 if (order_map_.contains(order_id)) {
130 return false;
131 }
132 // Use timestamp as priority if not specified
133 uint64_t actual_priority = (priority == 0) ? timestamp : priority;
134 return addOrModifyOrder(order_id, side, price, quantity, timestamp, actual_priority, seq_num, is_last_in_batch);
135}

137SLICK_OB_INLINE bool OrderBookL3::modifyOrder(OrderId order_id, Price new_price, Quantity new_quantity,
138 uint64_t seq_num, bool is_last_in_batch) {
139 // Validate sequence number - reject out-of-order (seq_num < last_seq_num)
140 if (seq_num > 0) {
141 if (seq_num < last_seq_num_) {
142 // Out of order - reject
143 return false;
144 }
145 last_seq_num_ = seq_num;
146 }
147
148 // Find order
149 detail::Order* order = order_map_.find(order_id);
150 if (!order) {
151 return false;
152 }
153
154 if (SLICK_UNLIKELY(new_quantity < 0)) {
155 return false;
156 }
157
158 // Check if this is a delete (quantity = 0)
159 if (new_quantity == 0) {
160 return deleteOrder(order_id, seq_num, is_last_in_batch);
161 }
162
163 const Timestamp timestamp = order->timestamp; // Use original timestamp for modifications
164 const Price old_price = order->price;
165 const Quantity old_quantity = order->quantity;
166 const Side side = order->side;
167
168 // Check what changed
169 const bool price_changed = (new_price != old_price);
170 const bool quantity_changed = (new_quantity != old_quantity);
171
172 if (!price_changed && !quantity_changed) {
173 // No change
174 return true;
175 }
176
177 if (price_changed) {
178 // Price changed - remove from old level (if exists) and add to new level
179 auto [old_level, old_level_idx] = getLevel(side, old_price);
180 Quantity old_level_total = 0;
181
182 if (old_level) {
183 // Remove from old level
184 old_level->removeOrder(order);
185 old_level_total = old_level->getTotalQuantity();
186
187 uint8_t old_level_change_flags = QuantityChanged;
188 if (removeLevelIfEmpty(side, old_price)) {
189 old_level_change_flags |= PriceChanged;
190 }
191 // Don't add LastInBatch to intermediate old level update
192 notifyPriceLevelUpdate(side, old_price, old_level_total, timestamp, old_level_idx, old_level_change_flags, seq_num);
193 }
194
195 // Update order fields
196 order->price = new_price;
197 order->quantity = new_quantity;
198
199 // Get or create new level
200 auto [new_level, new_level_idx, is_new] = getOrCreateLevel(side, new_price);
201
202 // Insert into new level
203 new_level->insertOrder(order);
204
205 // Notify observers (price changed: both price and quantity flags)
206 uint8_t order_flags = PriceChanged;
207 if (quantity_changed) {
208 order_flags |= QuantityChanged;
209 }
210 uint8_t new_level_change_flags = QuantityChanged;
211 if (is_new) {
212 new_level_change_flags |= PriceChanged;
213 }
214 // Add LastInBatch flag if this is the last update
215 if (is_last_in_batch) {
216 order_flags |= LastInBatch;
217 new_level_change_flags |= LastInBatch;
218 }
219
220 notifyOrderUpdate(order, old_quantity, old_price, timestamp, new_level_idx, order_flags, seq_num);
221 notifyPriceLevelUpdate(side, new_price, new_level->getTotalQuantity(), timestamp, new_level_idx, new_level_change_flags, seq_num);
222 notifyTopOfBookIfChanged(timestamp, order_flags);
223
224 } else {
225 // Only quantity changed
226 auto [level, level_index, is_new] = getOrCreateLevel(side, old_price);
227
228 // Update quantity
229 level->updateOrderQuantity(old_quantity, new_quantity);
230 order->quantity = new_quantity;
231
232 // Notify observers (only quantity changed)
233 uint8_t order_flags = QuantityChanged;
234 uint8_t level_change_flags = QuantityChanged;
235 if (is_new) [[unlikely]] {
236 level_change_flags |= PriceChanged;
237 }
238 // Add LastInBatch flag if this is the last update
239 if (is_last_in_batch) {
240 order_flags |= LastInBatch;
241 level_change_flags |= LastInBatch;
242 }
243
244 notifyOrderUpdate(order, old_quantity, old_price, timestamp, level_index, order_flags, seq_num);
245 notifyPriceLevelUpdate(side, old_price, level->getTotalQuantity(), timestamp, level_index, level_change_flags, seq_num);
246 notifyTopOfBookIfChanged(timestamp, order_flags);
247 }
248
249 return true;
250}

252SLICK_OB_INLINE bool OrderBookL3::deleteOrder(OrderId order_id, uint64_t seq_num, bool is_last_in_batch) noexcept {
253 // Validate sequence number - reject out-of-order (seq_num < last_seq_num)
254 if (seq_num > 0) {
255 if (seq_num < last_seq_num_) {
256 // Out of order - reject
257 return false;
258 }
259 last_seq_num_ = seq_num;
260 }
261
262 // Find order
263 detail::Order* order = order_map_.find(order_id);
264 if (!order) {
265 return false;
266 }
267
268 const Timestamp timestamp = order->timestamp;
269 const Price price = order->price;
270 const Side side = order->side;
271
272 // Get level
273 auto [level, level_idx] = getLevel(side, price);
274 if (!level) {
275 // Level doesn't exist - data structure inconsistency
276 // Deletion: both price and quantity changed
277 uint8_t order_flags = PriceChanged | QuantityChanged;
278 if (is_last_in_batch) {
279 order_flags |= LastInBatch;
280 }
281 // Notify deletion, then clean up (use max index since level not found)
282 notifyOrderDelete(order, timestamp, std::numeric_limits<uint16_t>::max(), order_flags, seq_num);
283 order_map_.erase(order_id);
284 order_pool_.destroy(order);
285 return false;
286 }
287
288 // Remove from level
289 level->removeOrder(order);
290 const Quantity level_total = level->getTotalQuantity();
291
292 // Remove from order map
293 order_map_.erase(order_id);
294
295 // Deletion: both price and quantity changed
296 uint8_t order_flags = PriceChanged | QuantityChanged;
297 uint8_t level_change_flags = QuantityChanged;
298 if (removeLevelIfEmpty(side, price)) {
299 level_change_flags |= PriceChanged;
300 }
301 // Add LastInBatch flag if this is the last update
302 if (is_last_in_batch) {
303 order_flags |= LastInBatch;
304 level_change_flags |= LastInBatch;
305 }
306
307 // Notify observers (before destroying order)
308 notifyOrderDelete(order, timestamp, level_idx, order_flags, seq_num);
309 notifyPriceLevelUpdate(side, price, level_total, timestamp, level_idx, level_change_flags, seq_num);
310
311 // Destroy order
312 order_pool_.destroy(order);
313
314 // Notify ToB if changed
315 notifyTopOfBookIfChanged(timestamp, order_flags);
316
317 return true;
318}

320SLICK_OB_INLINE bool OrderBookL3::executeOrder(OrderId order_id, Quantity executed_quantity, uint64_t seq_num, bool is_last_in_batch) {
321 // Validate sequence number - reject out-of-order (seq_num < last_seq_num)
322 if (seq_num > 0) {
323 if (seq_num < last_seq_num_) {
324 // Out of order - reject
325 return false;
326 }
327 last_seq_num_ = seq_num;
328 }
329
330 detail::Order* order = order_map_.find(order_id);
331 if (!order) {
332 return false;
333 }
334
335 if (SLICK_UNLIKELY(executed_quantity <= 0 || executed_quantity > order->quantity)) {
336 SLICK_ASSERT(executed_quantity > 0 && executed_quantity <= order->quantity);
337 return false;
338 }
339
340 const Quantity remaining = order->quantity - executed_quantity;
341
342 if (remaining == 0) {
343 // Fully executed - delete order (pass through seq_num and is_last_in_batch)
344 return deleteOrder(order_id, seq_num, is_last_in_batch);
345 } else {
346 // Partial execution - reduce quantity (pass through seq_num and is_last_in_batch)
347 return modifyOrder(order_id, order->price, remaining, seq_num, is_last_in_batch);
348 }
349}

355SLICK_OB_INLINE const detail::PriceLevelL3* OrderBookL3::getBestBid() const noexcept {
356 const auto& bids = levels_[Side::Buy];
357 if (bids.empty()) {
358 return nullptr;
359 }
360 // FlatMap is sorted by key (Price) in ascending order by default
361 // For Buy side, we want highest price first, so use reverse iterator
362 return &bids.begin()->second;
363}

366 const auto& asks = levels_[Side::Sell];
367 if (asks.empty()) {
368 return nullptr;
369 }
370 // For Sell side, we want lowest price first (first element)
371 return &asks.begin()->second;

374SLICK_OB_INLINE TopOfBook OrderBookL3::getTopOfBook() const noexcept {
375 const auto* bid = getBestBid();
376 const auto* ask = getBestAsk();
377
378 TopOfBook tob;
379 tob.symbol = symbol_;
380
381 if (bid) {
382 tob.best_bid = bid->price;
383 tob.bid_quantity = bid->getTotalQuantity();
384 if (!bid->orders.empty()) {
385 tob.timestamp = bid->orders.front()->timestamp;
386 }
387 }
388
389 if (ask) {
390 tob.best_ask = ask->price;
391 tob.ask_quantity = ask->getTotalQuantity();
392 if (!ask->orders.empty()) {
393 const Timestamp ask_time = ask->orders.front()->timestamp;
394 if (ask_time > tob.timestamp) {
395 tob.timestamp = ask_time;
396 }
397 }
398 }
399
400 return tob;
401}

403SLICK_OB_INLINE std::vector<detail::PriceLevelL2> OrderBookL3::getLevelsL2(Side side, std::size_t depth) const {
404 SLICK_ASSERT(side < SideCount);
405
406 const auto& level_map = levels_[side];
407 std::vector<detail::PriceLevelL2> result;
408
409 if (level_map.empty()) {
410 return result;
411 }
412
413 const std::size_t count = (depth == 0) ? level_map.size() : std::min(depth, level_map.size());
414 result.reserve(count);
415
416 if (side == Side::Buy) {
417 // For bids, iterate in reverse (highest price first)
418 auto it = level_map.begin();
419 for (std::size_t i = 0; i < count && it != level_map.end(); ++i, ++it) {
420 const auto& [price, level] = *it;
421 Timestamp latest_timestamp = level.orders.empty() ? 0 : level.orders.front()->timestamp;
422 result.emplace_back(price, level.getTotalQuantity(), latest_timestamp);
423 }
424 } else {
425 // For asks, iterate forward (lowest price first)
426 auto it = level_map.begin();
427 for (std::size_t i = 0; i < count && it != level_map.end(); ++i, ++it) {
428 const auto& [price, level] = *it;
429 Timestamp latest_timestamp = level.orders.empty() ? 0 : level.orders.front()->timestamp;
430 result.emplace_back(price, level.getTotalQuantity(), latest_timestamp);
431 }
432 }
433
434 return result;
435}

493SLICK_OB_INLINE void OrderBookL3::clearSide(Side side) noexcept {
494 SLICK_ASSERT(side < SideCount);
495
496 // Delete all orders on this side
497 auto& level_map = levels_[side];
498 for (auto& [price, level] : level_map) {
499 for (auto it = level.orders.begin(); it != level.orders.end(); ) {
500 auto* order = &(*it);
501 ++it; // Advance before unlinking/destroying.
502 order_map_.erase(order->order_id);
503 level.orders.erase(order);
504 order_pool_.destroy(order);
505 }
506 }
507
508 // Clear level map
509 level_map.clear();
510}

517SLICK_OB_INLINE std::tuple<detail::PriceLevelL3*, uint16_t, bool> OrderBookL3::getOrCreateLevel(Side side, Price price) {
518 SLICK_ASSERT(side < SideCount);
519
520 auto& level_map = levels_[side];
521 auto it = level_map.find(price);
522
523 if (it != level_map.end()) {
524 uint16_t index = static_cast<uint16_t>(std::distance(level_map.begin(), it));
525 return {&it->second, index, false};
526 }
527
528 // Create new level
529 auto [new_it, inserted] = level_map.emplace(price, detail::PriceLevelL3{price});
530 SLICK_ASSERT(inserted);
531 uint16_t index = static_cast<uint16_t>(std::distance(level_map.begin(), new_it));
532 return {&new_it->second, index, true};
533}

535SLICK_OB_INLINE bool OrderBookL3::removeLevelIfEmpty(Side side, Price price) noexcept {
536 SLICK_ASSERT(side < SideCount);
537
538 auto& level_map = levels_[side];
539 auto it = level_map.find(price);
540
541 if (it != level_map.end() && it->second.isEmpty()) {
542 level_map.erase(it);
543 return true;
544 }
545 return false;

548SLICK_OB_INLINE uint16_t OrderBookL3::calculateLevelIndex(Side side, Price price) const noexcept {
549 SLICK_ASSERT(side < SideCount);
550 const auto& level_map = levels_[side];
551
552 // Find the price level using lower_bound
553 auto it = level_map.lower_bound(price);
554
555 // If exact match, calculate its index
556 if (it != level_map.end() && it->first == price) {
557 return static_cast<uint16_t>(std::distance(level_map.begin(), it));
558 }
559
560 // Price not found - return the index where it would be inserted
561 return static_cast<uint16_t>(std::distance(level_map.begin(), it));
562}

629SLICK_OB_INLINE void OrderBookL3::notifyTopOfBookIfChanged(Timestamp timestamp, uint8_t update_flags) {
630 // Only emit ToB if LastInBatch flag is set
631 // (defaults to true for single operations, false for intermediate batch updates)
632 if ((update_flags & LastInBatch) == 0) {
633 return; // Skip ToB emission for intermediate batch updates
634 }
635
636 // Compute current top-of-book
637 const auto* bid = getBestBid();
638 const auto* ask = getBestAsk();
639
640 Price new_best_bid = bid ? bid->price : 0;
641 Quantity new_bid_qty = bid ? bid->getTotalQuantity() : 0;
642 Price new_best_ask = ask ? ask->price : 0;
643 Quantity new_ask_qty = ask ? ask->getTotalQuantity() : 0;
644
645 // Check if best bid or ask changed
646 const bool bid_changed = (cached_tob_.best_bid != new_best_bid) ||
647 (cached_tob_.bid_quantity != new_bid_qty);
648 const bool ask_changed = (cached_tob_.best_ask != new_best_ask) ||
649 (cached_tob_.ask_quantity != new_ask_qty);
650
651 if (bid_changed || ask_changed) {
652 // Update cached values
653 cached_tob_.best_bid = new_best_bid;
654 cached_tob_.bid_quantity = new_bid_qty;
655 cached_tob_.best_ask = new_best_ask;
656 cached_tob_.ask_quantity = new_ask_qty;
657 cached_tob_.timestamp = timestamp;
658
659 // Notify observers
660 observers_.notifyTopOfBookUpdate(cached_tob_);
661 }
662}

664SLICK_OB_INLINE void OrderBookL3::emitSnapshot(Timestamp timestamp) {
665 observers_.notifySnapshotBegin(symbol_, last_seq_num_, timestamp);
666
667 // Bids: highest price first (best = index 0)
668 uint16_t level_idx = 0;
669 for (auto it = levels_[Side::Buy].begin(); it != levels_[Side::Buy].end(); ++it, ++level_idx) {
670 const auto& level = it->second;
671 for (const auto& order : level.orders) {
672 OrderUpdate update{
673 symbol_,
674 order.order_id,
675 order.side,
676 order.price,
677 order.quantity,
678 timestamp,
679 level_idx,
680 order.priority,
681 static_cast<uint8_t>(PriceChanged | QuantityChanged)
682 };
683 observers_.notifyOrderUpdate(update);
684 }
685 }
686
687 // Asks: lowest price first (best = index 0)
688 level_idx = 0;
689 for (auto it = levels_[Side::Sell].begin(); it != levels_[Side::Sell].end(); ++it, ++level_idx) {
690 const auto& level = it->second;
691 for (const auto& order : level.orders) {
692 OrderUpdate update{
693 symbol_,
694 order.order_id,
695 order.side,
696 order.price,
697 order.quantity,
698 timestamp,
699 level_idx,
700 order.priority,
