48 void insertOrder(Order* order) noexcept {
49 // Insert in priority order (lower priority value = higher priority)
50 auto it = orders.begin();
51 while (it != orders.end() && it->priority <= order->priority) {
52 ++it;
53 }
54
55 if (it == orders.end()) {
56 orders.push_back(order);
57 } else {
58 orders.insert_before(&(*it), order);
59 }
60
61 total_quantity += order->quantity;
62 }

65 void removeOrder(Order* order) noexcept {
66 total_quantity -= order->quantity;
67 orders.erase(order);
68 }
