102 std::pair<iterator, bool> insertOrUpdate(Price price, Quantity quantity, Timestamp timestamp) {
103 // Binary search for insertion position
104 PriceLevelL2 key{price, 0, 0};
105 auto it = std::lower_bound(levels_.begin(), levels_.end(), key, compare_);
106
107 // Check if price already exists
108 if (it != levels_.end() && it->price == price) {
109 // Update existing level
110 it->quantity = quantity;
111 it->timestamp = timestamp;
112 return {it, false};
113 }
114
115 // Insert new level
116 it = levels_.insert(it, PriceLevelL2{price, quantity, timestamp});
117 return {it, true};
118 }
