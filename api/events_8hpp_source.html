142struct TopOfBook {
143 SymbolId symbol; // Symbol identifier
144 Price best_bid; // Best bid price (highest buy price)
145 Quantity bid_quantity; // Total quantity at best bid
146 Price best_ask; // Best ask price (lowest sell price)
147 Quantity ask_quantity; // Total quantity at best ask
148 Timestamp timestamp; // Snapshot timestamp
149 std::array<uint8_t, 2> change_flags {0, 0}; // Bitset of ChangeFlag
150
151 TopOfBook() noexcept
152 : symbol(0), best_bid(0), bid_quantity(0), best_ask(0), ask_quantity(0), timestamp(0) {}
153
154 TopOfBook(SymbolId sym, Price bid, Quantity bid_qty, Price ask, Quantity ask_qty, Timestamp ts) noexcept
155 : symbol(sym), best_bid(bid), bid_quantity(bid_qty), best_ask(ask), ask_quantity(ask_qty), timestamp(ts) {}
156
158 [[nodiscard]] Price spread() const noexcept {
159 return (best_ask > best_bid) ? (best_ask - best_bid) : 0;
160 }
161
163 [[nodiscard]] Price midPrice() const noexcept {
164 return (best_bid + best_ask) / 2;
165 }
166
168 [[nodiscard]] bool isCrossed() const noexcept {
169 return best_bid >= best_ask && best_bid > 0 && best_ask > 0;
170 }
171
173 [[nodiscard]] bool hasBid() const noexcept {
174 return best_bid > 0 && bid_quantity > 0;
175 }
176
178 [[nodiscard]] bool hasAsk() const noexcept {
179 return best_ask > 0 && ask_quantity > 0;
180 }
181
183 [[nodiscard]] bool isValid() const noexcept {
184 return hasBid() && hasAsk();
185 }
186};
