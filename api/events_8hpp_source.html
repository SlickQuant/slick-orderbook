138struct TopOfBook {
139 SymbolId symbol; // Symbol identifier
140 Price best_bid; // Best bid price (highest buy price)
141 Quantity bid_quantity; // Total quantity at best bid
142 Price best_ask; // Best ask price (lowest sell price)
143 Quantity ask_quantity; // Total quantity at best ask
144 Timestamp timestamp; // Snapshot timestamp
145
146 TopOfBook() noexcept
147 : symbol(0), best_bid(0), bid_quantity(0), best_ask(0), ask_quantity(0), timestamp(0) {}
148
149 TopOfBook(SymbolId sym, Price bid, Quantity bid_qty, Price ask, Quantity ask_qty, Timestamp ts) noexcept
150 : symbol(sym), best_bid(bid), bid_quantity(bid_qty), best_ask(ask), ask_quantity(ask_qty), timestamp(ts) {}
151
153 [[nodiscard]] Price spread() const noexcept {
154 return (best_ask > best_bid) ? (best_ask - best_bid) : 0;
155 }
156
158 [[nodiscard]] Price midPrice() const noexcept {
159 return (best_bid + best_ask) / 2;
160 }
161
163 [[nodiscard]] bool isCrossed() const noexcept {
164 return best_bid >= best_ask && best_bid > 0 && best_ask > 0;
165 }
166
168 [[nodiscard]] bool hasBid() const noexcept {
169 return best_bid > 0 && bid_quantity > 0;
170 }
171
173 [[nodiscard]] bool hasAsk() const noexcept {
174 return best_ask > 0 && ask_quantity > 0;
175 }
176
178 [[nodiscard]] bool isValid() const noexcept {
179 return hasBid() && hasAsk();
180 }
181};
