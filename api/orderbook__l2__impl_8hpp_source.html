89SLICK_OB_INLINE void OrderBookL2::updateLevel(Side side, Price price, Quantity quantity, Timestamp timestamp,
90 uint64_t seq_num, bool is_last_in_batch) {
91 SLICK_ASSERT(side < SideCount);
92
93 // Validate sequence number - reject out-of-order (seq_num < last_seq_num)
94 if (seq_num > 0) {
95 if (seq_num < last_seq_num_) {
96 // Out of order - reject silently
97 return;
98 }
99 last_seq_num_ = seq_num;
100 }
101
102 if (quantity == 0) {
103 // Find level index before deletion
104 auto it = sides_[side].find(price);
105 if (it != sides_[side].end()) {
106 uint16_t level_idx = static_cast<uint16_t>(std::distance(sides_[side].begin(), it));
107
108 // track starting index
109 change_starting_index_ = std::min<uint16_t>(change_starting_index_, level_idx);
110
111 // Delete the level
112 sides_[side].erase(it);
113
114 // Deletion: both price and quantity changed
115 uint8_t change_flags = PriceChanged | QuantityChanged;
116 if (is_last_in_batch) {
117 change_flags |= LastInBatch;
118 }
119
120 // Notify with level_index, flags, and seq_num
121 PriceLevelUpdate update{symbol_, side, price, 0, timestamp, level_idx, change_flags, seq_num};
122 observers_.notifyPriceLevelUpdate(update);
123 if (change_starting_index_ == 0 && is_last_in_batch) {
124 notifyTopOfBookIfChanged(timestamp);
125 change_starting_index_ = INVALID_INDEX;
126 }
127 }
128 } else {
129 // Insert or update level
130 auto [it, inserted] = sides_[side].insertOrUpdate(price, quantity, timestamp);
131
132 // Calculate level index
133 uint16_t level_idx = static_cast<uint16_t>(std::distance(sides_[side].begin(), it));
134
135 // track starting index
136 change_starting_index_ = std::min<uint16_t>(change_starting_index_, level_idx);
137
138 // Determine change flags based on what actually changed
139 uint8_t change_flags = 0;
140 if (inserted) {
141 // New level: both price and quantity changed
142 change_flags = PriceChanged | QuantityChanged;
143 } else {
144 // Existing level: only quantity changed
145 change_flags = QuantityChanged;
146 }
147
148 // Add LastInBatch flag if this is the last update
149 if (is_last_in_batch) {
150 change_flags |= LastInBatch;
151 }
152
153 // Notify observers
154 PriceLevelUpdate update{symbol_, side, price, quantity, timestamp, level_idx, change_flags, seq_num};
155 observers_.notifyPriceLevelUpdate(update);
156 if (change_starting_index_ == 0 && is_last_in_batch) {
157 notifyTopOfBookIfChanged(timestamp);
158 change_starting_index_ = INVALID_INDEX;
159 }
160 }
161}

178SLICK_OB_INLINE const detail::PriceLevelL2* OrderBookL2::getBestBid() const noexcept {
179 // Use sequence lock to read cached best bid atomically
180 // This is thread-safe for concurrent reads while a writer is updating
181 uint64_t seq1, seq2;
182 do {
183 seq1 = tob_seq_.load(std::memory_order_acquire);
184 // If seq1 is odd, writer is in progress, spin
185 while (seq1 & 1) {
186 seq1 = tob_seq_.load(std::memory_order_acquire);
187 }
188 detail::tsanAcquire(&tob_seq_);
189 // Check if there's a valid bid
190 if (cached_tob_.best_bid == 0) {
191 detail::tsanRelease(&tob_seq_);
192 return nullptr;
193 }
194 // Verify sequence didn't change during read
195 seq2 = tob_seq_.load(std::memory_order_acquire);
196 detail::tsanRelease(&tob_seq_);
197 } while (seq1 != seq2);
198
199 // Return pointer to cached value (const, read-only)
200 return &cached_best_bid_;
201}

228SLICK_OB_INLINE TopOfBook OrderBookL2::getTopOfBook() const noexcept {
229 // Use sequence lock to read cached top-of-book atomically
230 TopOfBook tob;
231 uint64_t seq1, seq2;
232 do {
233 seq1 = tob_seq_.load(std::memory_order_acquire);
234 // If seq1 is odd, writer is in progress, spin
235 while (seq1 & 1) {
236 seq1 = tob_seq_.load(std::memory_order_acquire);
237 }
238 detail::tsanAcquire(&tob_seq_);
239 // Read the cached value
240 tob = cached_tob_;
241 // Check if a write occurred during read
242 seq2 = tob_seq_.load(std::memory_order_acquire);
243 detail::tsanRelease(&tob_seq_);
244 } while (seq1 != seq2);
245 return tob;
246}

281SLICK_OB_INLINE void OrderBookL2::notifyTopOfBookIfChanged(Timestamp timestamp) {
282 // Compute current top-of-book (direct access, writer-side only)
283 const auto* bid = sides_[Side::Buy].best();
284 const auto* ask = sides_[Side::Sell].best();
285
286 Price new_best_bid = bid ? bid->price : 0;
287 Quantity new_bid_qty = bid ? bid->quantity : 0;
288 Price new_best_ask = ask ? ask->price : 0;
289 Quantity new_ask_qty = ask ? ask->quantity : 0;
290
291 // Check if best bid or ask changed
292 const bool bid_changed = (cached_tob_.best_bid != new_best_bid) ||
293 (cached_tob_.bid_quantity != new_bid_qty);
294 const bool ask_changed = (cached_tob_.best_ask != new_best_ask) ||
295 (cached_tob_.ask_quantity != new_ask_qty);
296
297 if (bid_changed || ask_changed) {
298 detail::tsanAcquire(&tob_seq_);
299 // Update cached values using sequence lock (odd = writing, even = readable)
300 uint64_t seq = tob_seq_.load(std::memory_order_relaxed);
301 tob_seq_.store(seq + 1, std::memory_order_release); // Mark as writing (odd)
302
303 // Update cached TopOfBook
304 cached_tob_.best_bid = new_best_bid;
305 cached_tob_.bid_quantity = new_bid_qty;
306 cached_tob_.best_ask = new_best_ask;
307 cached_tob_.ask_quantity = new_ask_qty;
308 cached_tob_.timestamp = timestamp;
309
310 // Update cached best bid/ask levels for thread-safe access
311 if (bid) {
312 cached_best_bid_ = *bid;
313 } else {
314 cached_best_bid_ = detail::PriceLevelL2{}; // Clear
315 }
316 if (ask) {
317 cached_best_ask_ = *ask;
318 } else {
319 cached_best_ask_ = detail::PriceLevelL2{}; // Clear
320 }
321
322 tob_seq_.store(seq + 2, std::memory_order_release); // Mark as readable (even)
323 detail::tsanRelease(&tob_seq_);
324
325 // Notify observers
326 observers_.notifyTopOfBookUpdate(cached_tob_);
327 }
328}

330SLICK_OB_INLINE void OrderBookL2::emitSnapshot(Timestamp timestamp) {
331 // Notify snapshot begin
332 observers_.notifySnapshotBegin(symbol_, last_seq_num_, timestamp);
333
334 // Emit all bid levels
335 uint16_t level_idx = 0;
336 for (const auto& level : sides_[Side::Buy]) {
337 PriceLevelUpdate update{
338 symbol_,
339 Side::Buy,
340 level.price,
341 level.quantity,
342 timestamp,
343 level_idx++,
344 static_cast<uint8_t>(PriceChanged | QuantityChanged)
345 };
346 observers_.notifyPriceLevelUpdate(update);
347 }
348
349 // Emit all ask levels
350 level_idx = 0;
351 for (const auto& level : sides_[Side::Sell]) {
352 PriceLevelUpdate update{
353 symbol_,
354 Side::Sell,
355 level.price,
356 level.quantity,
357 timestamp,
358 level_idx++,
359 static_cast<uint8_t>(PriceChanged | QuantityChanged)
360 };
361 observers_.notifyPriceLevelUpdate(update);
362 }
363
364 // Notify snapshot end
365 observers_.notifySnapshotEnd(symbol_, last_seq_num_, timestamp);
366}
