25SLICK_OB_INLINE void OrderBookL2::updateLevel(Side side, Price price, Quantity quantity, Timestamp timestamp,
26 uint64_t seq_num, bool is_last_in_batch) {
27 SLICK_ASSERT(side < SideCount);
28
29 // Validate sequence number - reject out-of-order (seq_num < last_seq_num)
30 if (seq_num > 0) {
31 if (seq_num < last_seq_num_) {
32 // Out of order - reject silently
33 return;
34 }
35 last_seq_num_ = seq_num;
36 }
37
38 if (quantity == 0) {
39 // Find level index before deletion
40 auto it = sides_[side].find(price);
41 if (it != sides_[side].end()) {
42 uint16_t level_idx = static_cast<uint16_t>(std::distance(sides_[side].begin(), it));
43
44 // Delete the level
45 sides_[side].erase(it);
46
47 // Deletion: both price and quantity changed
48 uint8_t change_flags = PriceChanged | QuantityChanged;
49 if (is_last_in_batch) {
50 change_flags |= LastInBatch;
51 }
52
53 // Notify with level_index, flags, and seq_num
54 PriceLevelUpdate update{symbol_, side, price, 0, timestamp, level_idx, change_flags, seq_num};
55 observers_.notifyPriceLevelUpdate(update);
56 notifyTopOfBookIfChanged(timestamp, change_flags);
57 }
58 } else {
59 // Insert or update level
60 auto [it, inserted] = sides_[side].insertOrUpdate(price, quantity, timestamp);
61
62 // Calculate level index
63 uint16_t level_idx = static_cast<uint16_t>(std::distance(sides_[side].begin(), it));
64
65 // Determine change flags based on what actually changed
66 uint8_t change_flags = 0;
67 if (inserted) {
68 // New level: both price and quantity changed
69 change_flags = PriceChanged | QuantityChanged;
70 } else {
71 // Existing level: only quantity changed
72 change_flags = QuantityChanged;
73 }
74
75 // Add LastInBatch flag if this is the last update
76 if (is_last_in_batch) {
77 change_flags |= LastInBatch;
78 }
79
80 // Notify observers
81 PriceLevelUpdate update{symbol_, side, price, quantity, timestamp, level_idx, change_flags, seq_num};
82 observers_.notifyPriceLevelUpdate(update);
83 notifyTopOfBookIfChanged(timestamp, change_flags);
84 }
85}

110SLICK_OB_INLINE TopOfBook OrderBookL2::getTopOfBook() const noexcept {
111 const auto* bid = getBestBid();
112 const auto* ask = getBestAsk();
113
114 TopOfBook tob;
115 tob.symbol = symbol_;
116
117 if (bid) {
118 tob.best_bid = bid->price;
119 tob.bid_quantity = bid->quantity;
120 tob.timestamp = bid->timestamp;
121 }
122
123 if (ask) {
124 tob.best_ask = ask->price;
125 tob.ask_quantity = ask->quantity;
126 if (ask->timestamp > tob.timestamp) {
127 tob.timestamp = ask->timestamp;
128 }
129 }
130
131 return tob;
132}

153SLICK_OB_INLINE void OrderBookL2::notifyTopOfBookIfChanged(Timestamp timestamp, uint8_t update_flags) {
154 // Only emit ToB if LastInBatch flag is set
155 // (defaults to true for single operations, false for intermediate batch updates)
156 if ((update_flags & LastInBatch) == 0) {
157 return; // Skip ToB emission for intermediate batch updates
158 }
159
160 // Compute current top-of-book
161 const auto* bid = getBestBid();
162 const auto* ask = getBestAsk();
163
164 Price new_best_bid = bid ? bid->price : 0;
165 Quantity new_bid_qty = bid ? bid->quantity : 0;
166 Price new_best_ask = ask ? ask->price : 0;
167 Quantity new_ask_qty = ask ? ask->quantity : 0;
168
169 // Check if best bid or ask changed
170 const bool bid_changed = (cached_tob_.best_bid != new_best_bid) ||
171 (cached_tob_.bid_quantity != new_bid_qty);
172 const bool ask_changed = (cached_tob_.best_ask != new_best_ask) ||
173 (cached_tob_.ask_quantity != new_ask_qty);
174
175 if (bid_changed || ask_changed) {
176 // Update cached values
177 cached_tob_.best_bid = new_best_bid;
178 cached_tob_.bid_quantity = new_bid_qty;
179 cached_tob_.best_ask = new_best_ask;
180 cached_tob_.ask_quantity = new_ask_qty;
181 cached_tob_.timestamp = timestamp;
182
183 // Notify observers
184 observers_.notifyTopOfBookUpdate(cached_tob_);
185 }
186}

188SLICK_OB_INLINE void OrderBookL2::emitSnapshot(Timestamp timestamp) {
189 // Notify snapshot begin
190 observers_.notifySnapshotBegin(symbol_, last_seq_num_, timestamp);
191
192 // Emit all bid levels
193 uint16_t level_idx = 0;
194 for (const auto& level : sides_[Side::Buy]) {
195 PriceLevelUpdate update{
196 symbol_,
197 Side::Buy,
198 level.price,
199 level.quantity,
200 timestamp,
201 level_idx++,
202 static_cast<uint8_t>(PriceChanged | QuantityChanged)
203 };
204 observers_.notifyPriceLevelUpdate(update);
205 }
206
207 // Emit all ask levels
208 level_idx = 0;
209 for (const auto& level : sides_[Side::Sell]) {
210 PriceLevelUpdate update{
211 symbol_,
212 Side::Sell,
213 level.price,
214 level.quantity,
215 timestamp,
216 level_idx++,
217 static_cast<uint8_t>(PriceChanged | QuantityChanged)
218 };
219 observers_.notifyPriceLevelUpdate(update);
220 }
221
222 // Notify snapshot end
223 observers_.notifySnapshotEnd(symbol_, last_seq_num_, timestamp);
224}
