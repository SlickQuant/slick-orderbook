19template<typename OrderBookT>
20OrderBookT* OrderBookManager<OrderBookT>::getOrCreateOrderBook(SymbolId symbol) {
21 // First, try read-only access (shared lock) - common case
22 {
23 std::shared_lock lock(mutex_);
24 auto it = symbol_map_.find(symbol);
25 if (it != symbol_map_.end()) [[likely]] {
26 return it->second.get();
27 }
28 }
29
30 // Symbol doesn't exist - need exclusive lock to insert
31 std::unique_lock lock(mutex_);
32
33 // Double-check pattern: another thread might have inserted while we waited
34 auto it = symbol_map_.find(symbol);
35 if (it != symbol_map_.end()) {
36 return it->second.get();
37 }
38
39 // Create new orderbook
40 auto orderbook = std::make_unique<OrderBookT>(symbol);
41 OrderBookT* ptr = orderbook.get();
42 symbol_map_.emplace(symbol, std::move(orderbook));
43
44 return ptr;
45}
