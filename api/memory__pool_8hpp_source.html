43 explicit ObjectPool(std::size_t initial_capacity = 1024)

93 [[nodiscard]] T* allocate() {
94 // Try to get from free list first
95 if (free_list_ != nullptr) {
96 FreeNode* node = free_list_;
97 free_list_ = node->next;
98 ++size_;
99 return reinterpret_cast<T*>(node);
100 }
101
102 // No free objects, need to grow the pool
103 if (!grow()) {
104 return nullptr;
105 }
106
107 // Try again from free list
108 if (free_list_ != nullptr) {
109 FreeNode* node = free_list_;
110 free_list_ = node->next;
111 ++size_;
112 return reinterpret_cast<T*>(node);
113 }
114
115 return nullptr;
116 }

137 void deallocate(T* ptr) noexcept {
138 if (ptr == nullptr) {

142 // Add to free list

192 void clear() noexcept {
193 // Reset free list to include all objects from all blocks
194 free_list_ = nullptr;
195 for (Block& block : blocks_) {
196 for (std::size_t i = 0; i < block.count; ++i) {
197 T* obj = reinterpret_cast<T*>(
198 static_cast<char*>(block.memory) + i * sizeof(T));
199 FreeNode* node = reinterpret_cast<FreeNode*>(obj);
200 node->next = free_list_;
201 free_list_ = node;
202 }
203 }
204 size_ = 0;
205 }

207private:
209 struct FreeNode {
210 FreeNode* next;
211 };
212
213 static_assert(sizeof(FreeNode) <= sizeof(T),
214 "ObjectPool requires sizeof(T) >= sizeof(void*)");

217 struct Block {
218 void* memory; // Pointer to allocated memory
219 std::size_t count; // Number of objects in this block
220 };
221
225 bool grow(std::size_t min_count = 0) {
226 // Use exponential growth strategy
227 // Start with 64, double each time, up to 8192 per block
228 constexpr std::size_t kMinBlockSize = 64;
229 constexpr std::size_t kMaxBlockSize = 8192;
230
231 std::size_t block_size = kMinBlockSize;
232 if (!blocks_.empty()) {
233 block_size = std::min(blocks_.back().count * 2, kMaxBlockSize);
234 }
235
236 // Ensure we allocate at least min_count
237 if (block_size < min_count) {
238 block_size = min_count;
239 }
240
241 // Allocate new block
242 void* memory = ::operator new(block_size * sizeof(T),
243 std::align_val_t{alignof(T)},
244 std::nothrow);
245 if (memory == nullptr) {
246 return false;
247 }
248
249 // Add block to list
250 blocks_.push_back({memory, block_size});
251 capacity_ += block_size;
252
253 // Add all objects in this block to free list
254 for (std::size_t i = 0; i < block_size; ++i) {
255 T* obj = reinterpret_cast<T*>(
256 static_cast<char*>(memory) + i * sizeof(T));
257 FreeNode* node = reinterpret_cast<FreeNode*>(obj);
258 node->next = free_list_;
259 free_list_ = node;
260 }
261
262 return true;
263 }
264
265 std::vector<Block> blocks_; // List of allocated memory blocks
266 std::size_t capacity_; // Total capacity across all blocks
267 std::size_t size_; // Number of currently allocated objects
268 FreeNode* free_list_; // Head of free list
269};
